for max_depth in range(1,12):
    reg_tree = DecisionTreeRegressor(max_depth=max_depth)
    mse = -cross_val_score(reg_tree, predictors, outcome, cv=5, scoring='neg_mean_squared_error').mean()
    mse_score.append((max_depth, mse))
