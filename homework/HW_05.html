In [87]:
##Neural network

class NeuralNetwork(nn.Module):
    def __init__(self):
        super(NeuralNetwork, self).__init__()
        self.hidden = nn.Linear(len(predictor_columns), 10)  
        self.output = nn.Linear(10, 1)                      
        self.relu = nn.ReLU()                              

    def forward(self, x):
        x = self.relu(self.hidden(x))  
        x = self.output(x)             
        return x

model = NeuralNetwork()

criterion = nn.MSELoss()
optimizer = optim.Adam(model.parameters(), lr=0.01)

class ComplexNeuralNetwork(nn.Module):
    def __init__(self):
        super(ComplexNeuralNetwork, self).__init__()
        self.hidden1 = nn.Linear(len(predictor_columns), 32)  
        self.hidden2 = nn.Linear(32, 16)                     
        self.hidden3 = nn.Linear(16, 8)                      
        self.output = nn.Linear(8, 1)                        
        self.relu = nn.ReLU()                                
        self.leaky_relu = nn.LeakyReLU()   

    def forward(self, x):
        x = self.relu(self.hidden1(x))   
        x = self.leaky_relu(self.hidden2(x)) 
        x = self.relu(self.hidden3(x))  
        x = self.output(x)               
        return x
