#Standarize and thorch tensors
scaler = StandardScaler()

y_train_tensor = torch.tensor(y_train, dtype=torch.float32).view(-1, 1)
X_test_tensor = torch.tensor(X_test_scaled, dtype=torch.float32)

In [10]:
##Neural network

class NeuralNetwork(nn.Module):
    def __init__(self):
        super(NeuralNetwork, self).__init__()
        self.hidden = nn.Linear(len(predictor_columns), 10)  # Hidden layer with 10 nodes
        self.output = nn.Linear(10, 1)                      # Output layer
        self.relu = nn.ReLU()                               # ReLU activation

    def forward(self, x):
        x = self.relu(self.hidden(x))  # Apply ReLU
        x = self.output(x)             # Final output
        return x

model = NeuralNetwork()

criterion = nn.MSELoss()
optimizer = optim.Adam(model.parameters(), lr=0.01)

class ComplexNeuralNetwork(nn.Module):
    def __init__(self):
        super(ComplexNeuralNetwork, self).__init__()
        self.hidden1 = nn.Linear(len(predictor_columns), 32)  
        self.hidden2 = nn.Linear(32, 16)                     
        self.hidden3 = nn.Linear(16, 8)                      
        self.output = nn.Linear(8, 1)                        
        self.relu = nn.ReLU()                                
        self.leaky_relu = nn.LeakyReLU()   

    def forward(self, x):
        x = self.relu(self.hidden1(x))   # First hidden layer with ReLU
        x = self.leaky_relu(self.hidden2(x))  # Second hidden layer with Leaky ReLU
        x = self.relu(self.hidden3(x))   # Third hidden layer with ReLU
        x = self.output(x)               # Output layer
        return x
